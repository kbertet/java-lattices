13  /**
14   * This class provides a representation of couple (A,B) of objects.
15   *
16   * ![Couple](Couple.png)
17   *
18   * @uml Couple.png
19   * !include resources/fr/kbertet/util/Couple.iuml
20   *
21   * class Couple #LightCyan
22   * title Couple UML graph
23   * @author jeff
24   */
25  public class Couple {
26      /**
27       * Left hand side of this component.
28       */
29      private Object left;
30  
31      /**
32       * Right hand side of this component.
33       */
34      private Object right;
35  
36      /**
37       * Constructor from two objects.
38       *
39       * @param   l  left hand side of this component
40       * @param   r  right hand side of this component
41       */
42      public Couple(Object l, Object r) {
43          this.left = l;
44          this.right = r;
45      }
46  
47      /**
48       * Returns left hand side of this component.
49       *
50       * @return  left hand side of this component
51       */
52      public Object getLeft() {
53          return left;
54      }
55  
56      /**
57       * Set left hand side of this component.
58       *
59       * @param   left  hand side of this component
60       *
61       * @return  this for chaining
62       */
63      public Couple setLeft(Object left) {
64          this.left = left;
65          return this;
66      }
67  
68      /**
69       * Returns right hand side of this component.
70       *
71       * @return  right hand side of this component
72       */
73      public Object getRight() {
74          return right;
75      }
76  
77      /**
78       * Set right hand side of this component.
79       *
80       * @param   right  hand side of this component
81       *
82       * @return  this for chaining
83       */
84      public Couple setRight(Object right) {
85          this.right = right;
86          return this;
87      }
88  
89      /**
90       * Returns true if c is equals to this component.
91       *
92       * @param   c  Couple tested with this component
93       *
94       * @return  true if c is equals to this component.
95       */
96      public boolean equals(Couple c) {
97          return (this.left == c.getLeft()) && (this.right == c.getRight());
98      }
99  
100     /**
101      * Compute the hash code.
102      *
103      * @return  an integer representing the object
104      */
105     @Override
106     public int hashCode() {
107         return super.hashCode();
108     }
109 
110     /**
111      * Returns a string representations of this component.
112      *
113      * @return  a string representations of this component.
114      */
115     @Override
116     public String toString() {
117         return "(" + this.left.toString() + "," + this.right.toString() + ")";
118     }
119 }
