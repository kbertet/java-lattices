33      /**
34       * This class provides a representation of permutations.
35       *
36       * If this component transforms : 0 -> 1, 1 -> 0 & 2 -> 2.
37       * Then its length is 3 and
38       *
39       * The content contains :
40       *
41       * ~~~
42       * content[0]=1
43       * content[1]=0
44       * content[2]=2
45       * ~~~
46       */
47      private static class Permutation {
48          /**
49           * This component is a permutation of 0..length-1.
50           */
51          private int length;
52  
53          /**
54           * The transformation represented by this component.
55           *
56           * If this component transforms : 0 -> 1, 1 -> 0 & 2 -> 2.
57           * The field content contains :
58           *
59           * ~~~
60           * content[0]=1
61           * content[1]=0
62           * content[2]=2
63           * ~~~
64           */
65          private int[] content;
66  
67          /**
68           * Constructs identity of the set 0..n-1.
69           *
70           * @param   n  permutation of the set 0..n-1.
71           */
72          public Permutation(int n) {
73              this.length = n;
74              this.content = new int[n];
75              for (int i = 0; i < n; i++) {
76                  this.content[i] = i;
77              }
78          }
79  
80          /**
81           * Returns the transformation coded by this component.
82           *
83           * @return  the transformation coded by this component.
84           */
85          public int[] getContent() {
86              return this.content;
87          }
88  
89          /**
90           * Set the transformation coded by this component.
91           *
92           * Length of this component is update by this method.
93           *
94           * @param   c  the transformation coded in this component.
95           *
96           * @return  this for chaining
97           */
98          public Permutation setContent(int[] c) {
99              this.content = c;
100             this.length = c.length;
101             return this;
102         }
103 
104         /**
105          * Return length of this component.
106          *
107          * @return  length of this component.
108          */
109         public int getLength() {
110             return this.length;
111         }
112 
113         /**
114          * Set length of this componenet.
115          *
116          * @param   l  length of this component.
117          *
118          * @return  true if update is successful.
119          */
120         public boolean setLength(int l) {
121             if ((this.content.length == l) && (l <= this.getLength())) {
122                 this.length = l;
123                 return true;
124             } else {
125                 return false;
126             }
127         }
128 
129         /**
130          * Returns a string representation of this component.
131          *
132          * @return  a string representation of this component.
133          */
134         @Override
135         public String toString() {
136             String str = "";
137             for (int i = 0; i < this.length; i++) {
138                 str = str + this.content[i];
139             }
140             return str;
141         }
142 
143         /**
144          * Returns true if this component is equal to s.
145          *
146          * @param   s  test if this component is equal to s
147          *
148          * @return  true if this component is equal to s
149          */
150         public boolean equals(Permutation s) {
151             if (!(s.getLength() == this.length)) {
152                 return false;
153             } else {
154                 boolean tmp = true;
155                 for (int i = 0; i < this.length; i++) {
156                     tmp = tmp && (this.content[i] == s.getContent()[i]);
157                 }
158                 return tmp;
159             }
160         }
161 
162         /**
163          * Compute the hash code.
164          *
165          * @return  an integer representing the object
166          */
167         @Override
168         public int hashCode() {
169             return super.hashCode();
170         }
171     }

180     /**
181      * Returns a randomly generated lattice with nb nodes.
182      *
183      * @param   nb  Number of nodes in the randomly generated lattice
184      *
185      * @return  a randomly generated lattice with nb nodes
186      */
187     public static Lattice random(int nb) {
188         boolean done = false;
189         Lattice l = new Lattice();
190         while (!done) {
191             DAGraph dag = DAGraph.random(nb - 2); // what an ugly strategy :-(
192             Lattice tmp = new Lattice(dag);
193             Node top = new Node(new Integer(nb - 1));
194             tmp.addNode(top);
195             for (Node n : tmp.max()) {
196                 if (!n.equals(top)) {
197                     tmp.addEdge(n, top);
198                 }
199             }
200             Node bot = new Node(new Integer(nb));
201             tmp.addNode(bot);
202             for (Node n : tmp.min()) {
203                 if (!n.equals(bot)) {
204                     tmp.addEdge(bot, n);
205                 }
206             }
207             if (tmp.isLattice()) {
208                 done = true;
209                 l = tmp;
210             }
211         }
212         return l;
213     }

215     /**
216      * Returns the boolean algebra of cardinal 2^n.
217      *
218      * @param   n  cardinal of the boolean algebra return by this method is 2^n
219      *
220      * @return  the boolean algebra of cardinal 2^n
221      */
222     public static Lattice booleanAlgebra(int n) {
223         Lattice l = new Lattice();
224         BitSet b = new BitSet(n);
225         Node bot = new Node(b);
226         l.addNode(bot);
227         for (int i = 0; i < n; i++) {
228             BitSet bs = new BitSet(n);
229             bs.set(i, true);
230             Node next = new Node(bs);
231             l.addNode(next);
232             l.addEdge(bot, next);
233             recursiveBooleanAlgebra(next, l, n);
234         }
235         return l;
236     }

238     /**
239      * Computes successors of node n in the boolean algebra currently generated.
240      *
241      * @param   node  this method compute successors of this node
242      * @param   l     boolean algebra currently generated
243      * @param   n     the number of node of l will be 2^n at the end of computation
244      */
245     private static void recursiveBooleanAlgebra(Node node, Lattice l, int n) {
246         for (int i = 0; i < n; i++) {
247             BitSet b = (BitSet) node.getContent();
248             if (!b.get(i)) {
249                 BitSet bs = (BitSet) b.clone();
250                 bs.set(i, true);
251                 if (l.getNodeByContent(bs) == null) {
252                     Node next = new Node(bs);
253                     l.addNode(next);
254                     l.addEdge(node, next);
255                     recursiveBooleanAlgebra(next, l, n);
256                 } else {
257                     l.addEdge(node, l.getNodeByContent(bs));
258                 }
259             }
260         }
261     }

263     /**
264      * Returns the lattice of permutations of 1..n.
265      *
266      * Permutation are ordered as follows :
267      * A permutation s2 is a succesor of a permutation s1, if s2 is obtained from s1 by inverting two consecutive
268      * elements i and j such that before inversion j > i.
269      *
270      * Example : 124356 has following successors 214356, 142356, 124536 and 124365.
271      *
272      * The bottom of this lattice is identity (for exemple 123456) and the top is for instance 654321.
273      *
274      * @param   n  the lattice of permutations of the set 1..n
275      *
276      * @return  the lattice of permutations of 1..n.
277      */
278     public static Lattice permutationLattice(int n) {
279         Lattice l = new Lattice();
280         int[] content = new int[n];
281         for (int i = 0; i < n; i++) {
282             content[i] = i;
283         }
284         Permutation s = new Permutation(n);
285         s.setContent(content);
286         Node bot = new Node(s);
287         l.addNode(bot);
288         for (int i = 0; i < n - 1; i++) {
289             int[] newC = content.clone();
290             newC[i] = content[i + 1];
291             newC[i + 1] = content[i];
292             Permutation newS = new Permutation(n);
293             newS.setContent(newC);
294             Node succ = new Node(newS);
295             l.addNode(succ);
296             l.addEdge(bot, succ);
297             recursivePermutationLattice(succ, l, n);
298         }
299         return l;
300     }

302     /**
303      * Computes successors of node n in the lattice l.
304      *
305      * @param   node  successors of this node are computed by this method
306      * @param   l     lattice of permutations currently generated
307      * @param   n     lattice of permutation of the set 1..n is currently generated
308      */
309     private static void recursivePermutationLattice(Node node, Lattice l, int n) {
310         Permutation s = (Permutation) node.getContent();
311         for (int i = 0; i < s.getLength() - 1; i++) {
312             if (s.getContent()[i] < s.getContent()[i + 1]) {
313                 int[] newC = s.getContent().clone();
314                 Node currentNode = new Node();
315                 newC[i] = s.getContent()[i + 1];
316                 newC[i + 1] = s.getContent()[i];
317                 Permutation newP = new Permutation(n);
318                 newP.setContent(newC);
319                 boolean newNode = true;
320                 Iterator<Node> it = l.getNodes().iterator();
321                 while (it.hasNext() && newNode) {
322                     currentNode = it.next();
323                     Permutation currentContent = (Permutation) currentNode.getContent();
324                     newNode = !(currentContent.equals(newP));
325                 }
326                 if (newNode) {
327                     Permutation newS = new Permutation(n);
328                     newS.setContent(newC);
329                     Node next = new Node(newS);
330                     l.addNode(next);
331                     l.addEdge(node, next);
332                     recursivePermutationLattice(next, l, n);
333                 } else {
334                     l.addEdge(node, currentNode);
335                 }
336             }
337         }
338     }

340     /**
341      * Returns the lattice cartesian product of l and r.
342      *
343      * A node in the product is a cartesian product of two nodes
344      *
345      * There is an edge (n1, m1) -> (n2, m2) if and only if there are edges n1 -> n2 and m1 -> m2
346      *
347      * @param   l  Lattice of the left hand side of the product
348      * @param   r  Lattice of the right hand side of the product
349      *
350      * @return  the lattice cartesian product of l and r
351      */
352     public static Lattice product(Lattice l, Lattice r) {
353         Lattice prod = new Lattice();
354         // Create nodes
355         for (Node nL : l.getNodes()) {
356             for (Node nR : r.getNodes()) {
357                 prod.addNode(new Node(new Couple(nL.getContent(), nR.getContent())));
358             }
359         }
360         // Create edges
361         for (Node from : prod.getNodes()) {
362             for (Node to : prod.getNodes()) {
363                 if (l.containsEdge(l.getNodeByContent(((Couple) from.getContent()).getLeft()),
364                         l.getNodeByContent(((Couple) to.getContent()).getLeft()))
365                         && r.containsEdge(r.getNodeByContent(((Couple) from.getContent()).getRight()),
366                                 r.getNodeByContent(((Couple) to.getContent()).getRight()))) {
367                     prod.addEdge(from, to);
368                 }
369             }
370         }
371         return prod;
372     }
373     /**
374      * Returns lattice l in which convex c has been doubled.
375      *
376      * @param   l  a lattice
377      * @param   c  a convex subset of l, to be doubled.
378      *
379      * @return  a lattice construct from l by doubling the convex subset c.
380      */
381     public static Lattice doublingConvex(Lattice l, DAGraph c) {
382         Lattice doubled = new Lattice();
383         // Copy nodes by Content
384         for (Node n : l.getNodes()) {
385             if (c.containsNode(n)) {
386                 // These nodes are doubled
387                 Couple cpl0 = new Couple(n.getContent(), 0);
388                 Node n0 = new Node(cpl0);
389                 Couple cpl1 = new Couple(n.getContent(), 1);
390                 Node n1 = new Node(cpl1);
391                 doubled.addNode(n0);
392                 doubled.addNode(n1);
393             } else {
394                 // These nodes are just copied
395                 doubled.addNode(new Node(n.getContent()));
396             }
397         }
398         // Construct edges of doubled
399         Couple test = new Couple(0, 0); // used to test class of contents
400         for (Node x : doubled.getNodes()) {
401             for (Node y : doubled.getNodes()) {
402                 // Add an edge if x < y
403                 if (x.getContent().getClass() == test.getClass()) { // x was in convex c
404                     if (y.getContent().getClass() == test.getClass()) { // y was also in convex c
405                         // x & y were in convex c
406                         Couple cX = (Couple) x.getContent();
407                         Couple cY = (Couple) y.getContent();
408                         if ((cX.getLeft() == cY.getLeft()) && (((Integer) cX.getRight()).intValue() == 0)
409                                 && (((Integer) cY.getRight()).intValue() == 1)) {
410                             // Same content means same node. x is of the form (cX, 0) and y is of the for (cX, 1) so x < y in doubled.
411                             doubled.addEdge(x, y);
412                         } else {
413                             if (l.majorants(l.getNodeByContent(cX.getLeft())).contains(l.getNodeByContent(cY.getLeft()))
414                                     && (cX.getRight() == cY.getRight())) {
415                                 // x < y in l and x & y have the same second component si x < y in doubled.
416                                 doubled.addEdge(x, y);
417                             }
418                         }
419                     } else { // y wasn't in convex c
420                         // x was in c & y wasn't
421                         Couple cX = (Couple) x.getContent();
422                         if (l.majorants(l.getNodeByContent(cX.getLeft())).contains(l.getNodeByContent(y.getContent()))
423                                 && (((Integer) cX.getRight()).intValue() == 1)) {
424                             // x < y in l and second component of x is 1.
425                             doubled.addEdge(x, y);
426                         }
427                     }
428                 } else { // x wasn't in convex c
429                     if (y.getContent().getClass() == test.getClass()) { // y was in convex c
430                         // x wasn't in c but y was
431                         Couple cY = (Couple) y.getContent();
432                         if (l.majorants(l.getNodeByContent(x.getContent())).contains(l.getNodeByContent(cY.getLeft()))
433                                 && (((Integer) cY.getRight()).intValue() == 0)) {
434                                 // x < y in l and x & second component of y is 0.
435                                 doubled.addEdge(x, y);
436                             }
437                     } else { // y wasn't in convex c
438                         // x wasn't in c nor y
439                         if (l.majorants(l.getNodeByContent(x.getContent())).contains(l.getNodeByContent(y.getContent()))) {
440                                 // x < y in l and x & second component of y is 0.
441                                 doubled.addEdge(x, y);
442                             }
443                     }
444                 }
445             }
446         }
447         doubled.transitiveReduction();
448         return doubled;
449     }
