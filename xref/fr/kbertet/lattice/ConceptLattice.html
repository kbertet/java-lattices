32  /**
33   * This class extends class {@link Lattice} to provide specific methods
34   * to manipulate both a concept lattice or a closed set lattice.
35   *
36   * This class provides methods implementing classical operation on a concept lattice:
37   * join and meet reduction, concepts sets reduction, ...
38   *
39   * This class also provides two static method generating a concept lattice:
40   * methods {@link #diagramLattice} and {@link #completeLattice} both computes
41   * the closed set lattice of a given closure system.
42   * The firt one computes the hasse diagram of the closed set lattice
43   * by invoking  method {@link #immediateSuccessors}. This method implements  an
44   * adaptation of the well-known Bordat algorithm that also
45   * computes the dependance graph of the lattice where at once the minimal generators and the canonical
46   * direct basis of the lattice are encoded.
47   * The second static method computes the transitively closure of the lattice
48   * as the inclusion relation defined on all the closures
49   * generated by method {@link ClosureSystem#allClosures} that implements
50   * the well-known Wille algorithm.
51   *
52   * ![ConceptLattice](ConceptLattice.png)
53   *
54   * @uml ConceptLattice.png
55   * !include resources/fr/kbertet/dgraph/DAGraph.iuml
56   * !include resources/fr/kbertet/dgraph/DGraph.iuml
57   * !include resources/fr/kbertet/dgraph/Edge.iuml

232     /**
233      * Returns a clone of this component composed of a clone of each concept and each edge.
234      *
235      * @return  a concept lattice
236      */
237     @Override
238     public ConceptLattice clone() {
239         ConceptLattice conceptLattice = new ConceptLattice();
240         TreeMap<Concept, Concept> copy = new TreeMap<Concept, Concept>();
241         for (Node n : this.getNodes()) {
242             Concept c = (Concept) n;
243             Concept c2 = c.clone();
244             copy.put(c, c2);
245             conceptLattice.addNode(c2);
246         }
247         for (Edge ed : this.getEdges()) {
248             conceptLattice.addEdge(new Edge(copy.get(ed.getFrom()), copy.get(ed.getTo()), ed.getContent()));
249         }
250         return conceptLattice;
251     }

357     /**
358      * Replaces, if not empty, set A of each concept with the difference between itself
359      * and set A of its predecessors;
360      * Then replaces, if not empty, set B of each concept by
361      * the difference between itself and set B of its successors.
362      *
363      * @return  a boolean
364      *
365      * @todo  Comment the return
366      */
367     public boolean makeInclusionReduction() {
368         if (!this.containsConcepts()) {
369             return false;
370         }
371         boolean setA = this.containsAllSetA();
372         boolean setB = this.containsAllSetB();
373         if (!setA && !setB) {
374             return false;
375         }
376         // makes setA inclusion reduction
377         if (setA) {
378             // computation of an inverse topological sort
379             this.transpose();
380             ArrayList<Node> sort = this.topologicalSort();
381             this.transpose();
382             // reduction of set A
383             for (Node to : sort) {
384                 Concept cto = (Concept) to;
385                 for (Node from : this.getPredecessorNodes(to))  {
386                     Concept cfrom = (Concept) from;
387                     cto.getSetA().removeAll(cfrom.getSetA());
388                 }
389             }
390         }
391         // makes setB inclusion reduction
392         if (setB) {
393             // computation of a topological sort
394             ArrayList<Node> sort = this.topologicalSort();
395             // reduction of set B
396             for (Node to : sort) {
397                 Concept cto  = (Concept) to;
398                 for (Node from : this.getSuccessorNodes(to))  {
399                     Concept cfrom = (Concept) from;
400                     cto.getSetB().removeAll(cfrom.getSetB());
401                 }
402             }
403         }
404         return true;
405     }

407     /**
408      * Replaces set A of each join irreducible node by
409      * the difference between itself and set A of the unique predecessor.
410      *
411      * Others closed sets are replaced by an emptyset.
412      *
413      * @return  a boolean
414      *
415      * @todo  Comment the return
416      */
417     public boolean makeIrreduciblesReduction() {
418         // make inclusion reduction
419         if (this.makeInclusionReduction()) {
420             // check if not set A reduced concepts are join irreducibles
421             // and if not set B reduced concepts are meet irreducibles
422             TreeSet<Node> joinIrr = this.joinIrreducibles();
423             TreeSet<Node> meetIrr = this.meetIrreducibles();
424             for (Node n : this.getNodes()) {
425                 Concept c = (Concept) n;
426                 if (c.hasSetA() && !c.getSetA().isEmpty() && !joinIrr.contains(c)) {
427                     c.putSetA(new ComparableSet());
428                 }
429                 if (c.hasSetB() && !c.getSetB().isEmpty() && !meetIrr.contains(c)) {
430                     c.putSetB(new ComparableSet());
431                 }
432             }
433         }
434         return true;
435     }

438      * Returns a lattice where edges are valuated by the difference between
439      * set A of two adjacent concepts.
440      *
441      * @return  a boolean
442      *
443      * @todo  Change comment
444      */
445     public boolean makeEdgeValuation() {
446         if (!this.containsConcepts()) {
447             return false;
448         }
449         for (Node n1 : this.getNodes()) {
450             for (Edge ed : this.getSuccessorEdges(n1)) {
451                 if (!ed.hasContent()) {
452                     Node n2 = ed.getTo();
453                     TreeSet diff = new TreeSet();
454                     diff.addAll(((Concept) n2).getSetA());
455                     diff.removeAll(((Concept) n1).getSetA());
456                     ed.setContent(diff);
457                 }
458             }
459         }
460         return true;
461     }

465     /**
466      * Returns a lattice where join irreducibles node's content
467      * is replaced by the first element of set A.
468      *
469      * Other nodes are replaced by a new comparable.
470      *
471      * @return  a lattice
472      */
473     public Lattice getJoinReduction() {
474         if (!this.containsConcepts()) {
475             return null;
476         }
477         if (!this.containsAllSetA()) {
478             return null;
479         }
480         Lattice lattice = new Lattice();
481         //ConceptLattice csl = new ConceptLattice (this);
482         ConceptLattice csl = this.clone();
483         csl.makeIrreduciblesReduction();
484         TreeSet<Node> joinIrr = csl.joinIrreducibles();
485         // addition to lattice of a comparable issued from each reduced closed set
486         TreeMap<Node, Node> reduced = new TreeMap<Node, Node>();
487         for (Node n : csl.getNodes()) {
488             Concept c = (Concept) n;
489             Node nred;
490             if (c.hasSetA() && joinIrr.contains(n)) {
491                 nred = new Node(c.getSetA().first());
492             } else {
493                 nred = new Node();
494             }
495             reduced.put(n, nred);
496         }
497         // addtion of nodes to lattice
498         for (Node n : csl.getNodes()) {
499             lattice.addNode(reduced.get(n));
500         }
501         // addtion of edges to lattice
502         for (Node from : csl.getNodes()) {
503             for (Node to : csl.getSuccessorNodes(from)) {
504                 lattice.addEdge(reduced.get(from), reduced.get(to));
505             }
506         }
507         return lattice;
508     }

557     /**
558      * Returns a lattice where each join irreducible concept
559      * is replaced by a node containing the first element of set A,
560      * and each meet irreducible concept is replaced by a node contining the first element of set B.
561      *
562      * A concept that is at once join and meet irreducible is replaced by
563      * a node containing the first element of set A and the first element of set B in a set.
564      * Other nodes are replaced by an empty node.
565      *
566      * @return  a lattice
567      */
568     public Lattice getIrreduciblesReduction() {
569         Lattice lattice = new Lattice();
570         if (!this.containsConcepts()) {
571             return lattice;
572         }
573         //ConceptLattice csl = new ConceptLattice (this);
574         ConceptLattice csl = this.clone();
575         csl.makeIrreduciblesReduction();
576         TreeSet<Node> joinIrr = csl.joinIrreducibles();
577         TreeSet<Node> meetIrr = csl.meetIrreducibles();
578         // addition to lattice of a comparable issued from each reduced closed set
579         TreeMap<Node, Node> reduced = new TreeMap<Node, Node>();
580         for (Node n : csl.getNodes()) {
581             Concept c = (Concept) n;
582             // create a new Node with two indexed elements: the first of set A and the first of set B
583             if (c.hasSetA() && c.hasSetB() && meetIrr.contains(c) && joinIrr.contains(c)) {
584                 TreeSet<Comparable> content = new TreeSet<Comparable>();
585                 content.add(c.getSetA().first());
586                 content.add(c.getSetB().first());
587                 Node nred = new Node(content);
588                 reduced.put(n, nred);
589             } else if (c.hasSetA() && joinIrr.contains(n)) {
590                 // create a new Node with the first element of set A
591                 Node nred = new Node(((Concept) n).getSetA().first());
592                 reduced.put(n, nred);
593             } else if (c.hasSetB() && meetIrr.contains(n)) {
594                 // create a new Node with the first element of set A
595                 Node nred = new Node(((Concept) n).getSetB().first());
596                 reduced.put(n, nred);
597             } else {
598                 reduced.put(n, new Node());
599             }
600         }
601         // addtion of nodes to lattice
602         for (Node n : csl.getNodes()) {
603             lattice.addNode(reduced.get(n));
604         }
605         // addtion of edges to lattice
606         for (Node from : csl.getNodes()) {
607             for (Node to : csl.getSuccessorNodes(from)) {
608                 lattice.addEdge(reduced.get(from), reduced.get(to));
609             }
610         }
611         return lattice;
612     }

614     /**
615      * Generate the lattice composed of all the antichains of this component
616      * ordered with the inclusion relation.
617      *
618      * This treatment is performed in O(??) by implementation of Nourine algorithm
619      * that consists in a sequence of doubling intervals of nodes.
620      *
621      * @param   dag  a directed acyclic graph
622      *
623      * @return  the concept lattice
624      */
625     public static ConceptLattice idealLattice(DAGraph dag) {
626         if (!dag.isAcyclic()) {
627             return null;
628         }
629         // initialise the poset of ideals with the emptyset
630         ConceptLattice conceptLattice = new ConceptLattice();
631         int id = 1;
632         conceptLattice.addNode(new Concept(true, false));
633         // travel on graph according to a topological sort
634         DAGraph graph = new DAGraph(dag);
635         graph.transitiveClosure();
636         // treatment of nodes according to a topological sort
637         ArrayList<Node> sort = graph.topologicalSort();
638         for (Node x : sort) {
639             // computation of Jx
640             TreeSet<Node> jxmoins = new TreeSet<Node>(graph.getPredecessorNodes(x));
641             // storage of new ideals in a set
642             TreeSet<Concept> toAdd = new TreeSet<Concept>();
643             for (Node j1 : conceptLattice.getNodes()) {
644                 if (((Concept) j1).containsAllInA(jxmoins)) {
645                      Concept newJ = new Concept(true, false);
646                      newJ.addAllToA(((TreeSet) ((Concept) j1).getSetA()));
647                      newJ.addToA(x);
648                      toAdd.add(newJ);
649                 }
650             }
651             // addition of the new ideals in conceptLattice
652             for (Concept newJ : toAdd) {
653                 conceptLattice.addNode(newJ);
654             }
655         }
656         // computation of the inclusion relaton
657         for (Node node1 : conceptLattice.getNodes()) {
658             for (Node node2 : conceptLattice.getNodes()) {
659                 if (((Concept) node1).containsAllInA(((Concept) node2).getSetA())) {
660                     conceptLattice.addEdge(node2, node1);
661                 }
662             }
663         }
664         conceptLattice.transitiveReduction();
665         return conceptLattice;
666     }

668     /**
669      * Returns iceberg lattice whose concept contains enough observations.
670      *
671      * Are kept only concept whose number of observation is over threshold.
672      * A top node is added to keep the lattice structure.
673      *
674      * @param threshold used to determine nodes to be kept.
675      * @return iceberg lattice whose concept contains enough observations.
676      */
677     public ConceptLattice iceberg(float threshold) {
678         ConceptLattice l = new ConceptLattice();
679         Concept b = (Concept) this.bottom();
680         int card = b.getSetB().size();
681         for (Node n : this.getNodes()) {
682             Concept cpt = (Concept) n;
683             if ((float) cpt.getSetB().size() / (float) card >= threshold) {
684                 l.addNode(n);
685             }
686         }
687         for (Node f : l.getNodes()) {
688             for (Node t : l.getNodes()) {
689                 if (this.containsEdge(f, t)) {
690                     l.addEdge(f, t);
691                 }
692             }
693         }
694         Node t = this.top();
695         l.addNode(t);
696         for (Node n : l.getWells()) {
697             if (!n.equals(t)) {
698                 l.addEdge(n, t);
699             }
700         }
701         return l;
702     }

705     /**
706      * Generates and returns the complete (i.e. transitively closed) closed set lattice of the
707      * specified closure system, that can be an implicational system (ImplicationalSystem) or a context.
708      *
709      * The lattice is generated using the well-known Next Closure algorithm.
710      * All closures are first generated using the method:
711      * {@link ClosureSystem#allClosures}
712      * that implements the well-known Next Closure algorithm.
713      * Then, all concepts are ordered by inclusion.
714      *
715      * @param   init  a closure system (an ImplicationalSystem or a Context)
716      *
717      * @return  a concept lattice
718      */
719     public static ConceptLattice completeLattice(ClosureSystem init) {
720         ConceptLattice lattice = new ConceptLattice();
721         // compute all the closed set with allClosures
722         Vector<Concept> allclosure = init.allClosures();
723         for (Concept cl : allclosure) {
724             lattice.addNode(cl);
725         }
726 
727         // an edge corresponds to an inclusion between two closed sets
728         for (Node from : lattice.getNodes()) {
729             for (Node to : lattice.getNodes()) {
730                 if (((Concept) to).containsAllInA(((Concept) from).getSetA())) {
731                    lattice.addEdge(from, to);
732                 }
733             }
734         }
735         // Hasse diagram is computed
736         return lattice;
737     }

786     /**
787      * Returns the Hasse diagramme of the closed set lattice of the specified closure system
788      * issued from the specified concept.
789      *
790      * Immediate successors generation is an adaptation of Bordat's theorem
791      * stating that there is a bijection
792      * between minimal strongly connected component of the precedence subgraph issued
793      * from the specified node, and its immediate successors.
794      *
795      * This treatment is performed in O(cCl|S|^3log g) where S is the initial set of elements,
796      * c is the number of closed sets that could be exponential in the worst case,
797      * Cl is the closure computation complexity
798      * and g is the number of minimal generators of the lattice.
799      *
800      * @param   n     a concept
801      * @param   init  a closure system
802      */
803     public void recursiveDiagramLattice(Concept n, ClosureSystem init) {
804         Vector<TreeSet<Comparable>> immSucc = this.immediateSuccessors(n, init);
805         for (TreeSet<Comparable> setX : immSucc) {
806             Concept c = new Concept(new TreeSet(setX), false);
807             Concept ns = (Concept) this.getNode(c);
808             if (ns != null)  {
809                 // when ns already exists, addition of a new edge
810                 this.addEdge(n, ns);
811             } else { // when ns don't already exists, addition of a new node and recursive treatment
812                 this.addNode(c);
813                 this.addEdge(n, c);
814                 this.recursiveDiagramLattice(c, init);
815             }
816         }
817     }

819     /**
820      * Returns the list of immediate successors of a given node of the lattice.
821      *
822      * This treatment is an adaptation of Bordat's theorem stating that there is a bijection
823      * between minimal strongly connected component of the precedence subgraph issued
824      * from the specified node, and its immediate successors.
825      *
826      * This treatment is performed in O(Cl|S|^3log g) where S is the initial set of elements,
827      * Cl is the closure computation complexity
828      * and g is the number of minimal generators of the lattice.
829      *
830      * This treatment is recursively invoked by method recursiveDiagramlattice. In this case, the dependance graph
831      * is initialised by method recursiveDiagramMethod, and updated by this method,
832      * with addition some news edges and/or new valuations on existing edges.
833      * When this treatment is not invoked by method recursiveDiagramLattice, then the dependance graph
834      * is initialised, but it may be not complete. It is the case for example for on-line generation of the
835      * concept lattice.
836      *
837      * @param   n     a node
838      * @param   init  a closure system
839      *
840      * @return  a set of immediate successors
841      */
842     public Vector<TreeSet<Comparable>> immediateSuccessors(Node n, ClosureSystem init) {
843         // Initialisation of the dependance graph when not initialised by method recursiveDiagramLattice
844         if (!this.hasDependencyGraph()) {
845             DGraph graph = new DGraph();
846             for (Comparable c : init.getSet()) {
847                 graph.addNode(new Node(c));
848             }
849             this.setDependencyGraph(graph);
850         }
851         // computes newVal, the subset to be used to valuate every new dependance relation
852         // newVal = F\predecessors of F in the precedence graph of the closure system
853         // For a non reduced closure system, the precedence graph is not acyclic,
854         // and therefore strongly connected components have to be used.
855         ComparableSet setF = new ComparableSet(((Concept) n).getSetA());
856         DGraph prec = init.precedenceGraph();
857         DAGraph acyclPrec = prec.getStronglyConnectedComponent();
858         ComparableSet newVal = new ComparableSet();
859         newVal.addAll(setF);
860         for (Object x : setF)  {
861             // computes nx, the strongly connected component containing x
862             Node nx = null;
863             for (Node cc : acyclPrec.getNodes()) {
864                 TreeSet<Node> setCC = (TreeSet<Node>) cc.getContent();
865                 for (Node y : setCC) {
866                     if (x.equals(y.getContent())) {
867                         nx = cc;
868                     }
869                 }
870             }
871             // computes the minorants of nx in the acyclic graph
872             SortedSet<Node> ccMinNx = acyclPrec.minorants(nx);
873             // removes from newVal every minorants of nx
874             for (Node cc : ccMinNx) {
875                 TreeSet<Node> setCC = (TreeSet<Node>) cc.getContent();
876                 for (Node y : setCC) {
877                     newVal.remove(y.getContent());
878                 }
879             }
880         }
881         // computes the node belonging in S\F
882         TreeSet<Node> nodes = new TreeSet<Node>();
883         for (Node in : this.getDependencyGraph().getNodes()) {
884             if (!setF.contains(in.getContent())) {
885                 nodes.add(in);
886             }
887         }
888         // computes the dependance relation between nodes in S\F
889         // and valuated this relation by the subset of S\F
890         TreeSet<Edge> edges = new TreeSet<Edge>();
891         for (Node from : nodes) {
892             for (Node to : nodes) {
893                 if (!from.equals(to)) {
894                     // check if from is in dependance relation with to
895                     // i.e. "from" belongs to the closure of "F+to"
896                     ComparableSet fPlusTo = new ComparableSet(setF);
897                     fPlusTo.add(to.getContent());
898                     fPlusTo = new ComparableSet(init.closure(fPlusTo));
899                     if (fPlusTo.contains(from.getContent())) {
900                         // there is a dependance relation between from and to
901                         // search for an existing edge between from and to
902                         Edge ed = this.getDependencyGraph().getEdge(from, to);
903                         if (ed == null) {
904                             ed = new Edge(from, to, new TreeSet<ComparableSet>());
905                             this.getDependencyGraph().addEdge(ed);
906                         }
907                         edges.add(ed);
908                         // check if F is a minimal set closed for dependance relation between from and to
909                         ((TreeSet<ComparableSet>) ed.getContent()).add(newVal);
910                         TreeSet<ComparableSet> valEd = new TreeSet<ComparableSet>((TreeSet<ComparableSet>) ed.getContent());
911                         for (ComparableSet x1 : valEd) {
912                             if (x1.containsAll(newVal) && !newVal.containsAll(x1)) {
913                                 ((TreeSet<ComparableSet>) ed.getContent()).remove(x1);
914                             }
915                             if (!x1.containsAll(newVal) && newVal.containsAll(x1)) {
916                                 ((TreeSet<ComparableSet>) ed.getContent()).remove(newVal);
917                             }
918                         }
919                     }
920                 }
921             }
922         }
923         // computes the dependance subgraph of the closed set F as the reduction
924         // of the dependance graph composed of nodes in S\A and edges of the dependance relation
925         DGraph sub = this.getDependencyGraph().getSubgraphByNodes(nodes);
926         DGraph delta = sub.getSubgraphByEdges(edges);
927         // computes the sources of the CFC of the dependance subgraph
928         // that corresponds to successors of the closed set F
929         DAGraph cfc = delta.getStronglyConnectedComponent();
930         SortedSet<Node> sccMin = cfc.getSinks();
931         Vector<TreeSet<Comparable>> immSucc = new Vector<TreeSet<Comparable>>();
932         for (Node n1 : sccMin) {
933             TreeSet s = new TreeSet(setF);
934             TreeSet<Node> toadd = (TreeSet<Node>) n1.getContent();
935             for (Node n2 : toadd) {
936                 s.add(n2.getContent());
937             }
938             immSucc.add(s);
939         }
940        return immSucc;
941     }

943     /**
944      * Save the description of this component in a file whose name is specified.
945      *
946      * @param   filename  the name of the file
947      *
948      * @throws  IOException  When an IOException occurs
949      */
950     public void save(final String filename) throws IOException {
951         String extension = "";
952         int index = filename.lastIndexOf('.');
953         if (index > 0) {
954             extension = filename.substring(index + 1);
955         }
956         ConceptLatticeWriter writer = ConceptLatticeWriterFactory.get(extension);
957         if (writer == null) {
958             super.save(filename);
959         } else {
960             BufferedWriter file = new BufferedWriter(new FileWriter(filename));
961             writer.write(this, file);
962             file.close();
963         }
