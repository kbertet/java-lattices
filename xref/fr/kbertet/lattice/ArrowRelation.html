26  /**
27   * The ArrowRelation class encodes arrow relation between meet & join-irreductibles of a lattice.
28   *
29   * Let m and b be respectively meet and join irreductibles of a lattice.
30   * Recall that m has a unique successor say m+ and j has a unique predecessor say j-, then :
31   *
32   * - j "Up Arrow" m (stored as "Up") iff j is not less or equal than m and j is less than m+
33   * - j "Down Arrow" m (stored as "Down") iff j is not less or equal than m and j- is less than m
34   * - j "Up Down Arrow" m (stored as "UpDown") iff j "Up" m and j "Down" m
35   * - j "Cross" m (stored as "Cross") iff j is less or equal than m
36   * - j "Circ" m (stored as "Circ") iff neither j "Up" m nor j "Down" m nor j "Cross" m
37   *
38   * ![ArrowRelation](ArrowRelation.png)
39   *
40   * @uml ArrowRelation.png
41   * !include resources/fr/kbertet/dgraph/DGraph.iuml
42   * !include resources/fr/kbertet/dgraph/Edge.iuml
43   * !include resources/fr/kbertet/dgraph/Node.iuml
44   * !include resources/fr/kbertet/lattice/ArrowRelation.iuml
45   *
46   * hide members
47   * show ArrowRelation members
48   * class ArrowRelation #LightCyan
49   * title ArrowRelation UML graph
50   */

82      /**
83       * Unique constructor of this component from a lattice.
84       *
85       * Nodes are join or meet irreductibles of the lattice.
86       * Edges content encodes arrows as String "Up", "Down", "UpDown", "Cross", "Circ".
87       *
88       * @param   lattice  Lattice from which this component is deduced.
89       */
90      public ArrowRelation(Lattice lattice) {
91  
92          /* Nodes are join or meet irreductibles of the lattice. */
93          TreeSet<Node> joins = new TreeSet<Node>(lattice.joinIrreducibles());
94          for (Node n : joins) {
95              this.addNode(n);
96          }
97          TreeSet<Node> meets = new TreeSet<Node>(lattice.meetIrreducibles());
98          for (Node n : meets) {
99              this.addNode(n);
100         }
101         Lattice transitiveClosure = new Lattice(lattice);
102         transitiveClosure.transitiveClosure();
103         Lattice transitiveReduction = new Lattice(lattice);
104         transitiveReduction.transitiveReduction();
105         Node jminus = new Node();
106         Node mplus = new Node();
107         Object arrow = new Object();
108 
109         /* Content of edges are arrows */
110         for (Node j : joins) {
111             for (Node m : meets) {
112                 mplus = transitiveReduction.getSuccessorNodes(m).first();
113                 jminus = transitiveReduction.getPredecessorNodes(j).first();
114                 if (transitiveClosure.getSuccessorNodes(j).contains(m) || j.equals(m)) {
115                     arrow = ArrowRelation.cross;
116                 } else {
117                     if (transitiveClosure.getSuccessorNodes(jminus).contains(m) || jminus.equals(m)) {
118                         arrow = ArrowRelation.down;
119                         if (transitiveClosure.getPredecessorNodes(mplus).contains(j) || mplus.equals(j)) {
120                             arrow = ArrowRelation.updown;
121                         }
122                     } else {
123                         if (transitiveClosure.getPredecessorNodes(mplus).contains(j)) {
124                             arrow = ArrowRelation.up;
125                         } else {
126                             arrow = ArrowRelation.circ;
127                         }
128                     }
129                 }
130                 this.addEdge(j, m, arrow);
131             }
132         }
133     }

135     /**
136      * Save the description of this component in a file whose name is specified.
137      *
138      * @param   filename  the name of the file
139      *
140      * @throws  IOException  When an IOException occurs
141      */
142     public void save(final String filename) throws IOException {
143         String extension = "";
144         int index = filename.lastIndexOf('.');
145         if (index > 0) {
146             extension = filename.substring(index + 1);
147         }
148         BufferedWriter file = new BufferedWriter(new FileWriter(filename));
149         ArrowRelationWriterFactory.get(extension).write(this, file);
150         file.close();
151     }

153     /**
154      * Returns the table of the lattice, composed of the join and meet irreducibles nodes.
155      *
156      * Each attribute of the table is a copy of a join irreducibles node.
157      * Each observation of the table is a copy of a meet irreducibles node.
158      * An attribute is extent of an observation when its join irreducible node
159      * is in double arrow relation with the meet irreducible node in the lattice.
160      *
161      * @return  the table of the lattice
162      */
163     public Context getDoubleArrowTable() {
164         Context context = new Context();
165         // observations are join irreductibles
166         // attributes are meet irreductibles
167         for (Edge e : this.getEdges()) {
168             context.addToObservations(e.getFrom());
169             context.addToAttributes(e.getTo());
170         }
171         // generation of extent-intent
172         for (Edge e : this.getEdges()) {
173             if (e.getContent() == ArrowRelation.updown) {
174                 context.addExtentIntent(e.getFrom(), e.getTo());
175             }
176         }
177         return context;
178     }
