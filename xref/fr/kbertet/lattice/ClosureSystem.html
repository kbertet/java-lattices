25  /**
26   * This class is an abstract class defining the common behavior of closure systems,
27   * and specialy its closed set lattice generation.
28   *
29   * Both a context and an implicational system have properties of a closure system,
30   * and therefore extend this class.
31   *
32   * A closure system is formaly defined by a set of indexed elements and a closure operator
33   * (abstract methods {@link #getSet} and {@link #closure}).
34   *
35   * Abstract method {@link #save} also describe
36   * the common behavior of a closure system.
37   *
38   * However, this abstract class provides both abstract and non abstract methods.
39   *
40   * Although abstract methods depends on data, and so have to be implemented by each extended class,
41   * non abstract methods only used property of a closure system. It is the case for methods
42   * {@link #nextClosure} (that computes the next closure of the specified one according to the lectic order
43   * implemented the well-known Wille algorithm)
44   * invoked by method {@link #allClosures}
45   * and the main method {@link #closedSetLattice} (where lattice can be transitively closed or reduced).
46   *
47   *
48   * ![ClosureSystem](ClosureSystem.png)
49   *
50   * @uml ClosureSystem.png
51   * !include resources/fr/kbertet/lattice/ClosureSystem.iuml
52   *
53   * hide members
54   * show ClosureSystem members
55   * class ClosureSystem #LightCyan
56   * title ClosureSystem UML graph
57   */
58  public abstract class ClosureSystem {

89      /**
90       * Returns the closed set lattice of this component.
91       *
92       * A true value of the boolean `diagram` indicates that the
93       * Hasse diagramm of the lattice is computed (i.e. it is transitively reduced),
94       * whereas a false value indicates that the lattice is transitively closed
95       *
96       * A transitively reduced lattice is generated by the static method
97       * `ConceptLattice diagramLattice (ClosureSystem init)`
98       * that implements an adaptation of Bordat's algorithm.
99       * This adaptation computes the dependance graph while the lattice is generated,
100      * with the same complexity.
101      *
102      * A transitively closed lattice is generated bye well-known Next Closure algorithm.
103      * In this case, the dependance graph of the lattice isn't computed.
104      *
105      * @param   diagram  a boolean indicating if the Hasse diagramm of the lattice is computed or not.
106      *
107      * @return  The concept lattice
108      */
109     public ConceptLattice closedSetLattice(boolean  diagram) {
110         if (diagram) {
111             return ConceptLattice.diagramLattice(this);
112         } else {
113             return ConceptLattice.completeLattice(this);
114         }
115     }

117     /**
118      * Returns the lattice of this component.
119      *
120      * @return  The concept lattice
121      */
122     public ConceptLattice lattice() {
123         return this.closedSetLattice(true);
124     }

126    /**
127     * Returns all the closed sets of the specified closure system
128     * (that can be an IS or a context).
129     *
130     * Closed sets are generated in lecticaly order, with the emptyset's closure
131     * as first closed set, using the Ganter's Next Closure algorithm.
132     *
133     *Therefore,
134     * closed sets have to be comparable using `ComparableSet` class.
135     * This treatment is performed in O(cCl|S|^3) where S is the initial set of elements,
136     * c is the number of closed sets that could be exponential in the worst case,
137     * and Cl is the closure computation complexity.
138     *
139     * @return  all the closeds set in the lectically order.
140     */
141     public Vector<Concept> allClosures() {
142         Vector<Concept> allclosure = new Vector<Concept>();
143         // first closure: closure of the empty set
144         allclosure.add(new Concept(this.closure(new ComparableSet()), false));
145         Concept cl = allclosure.firstElement();
146         // next closures in lectically order
147         boolean continu = true;
148         do {
149             cl = this.nextClosure(cl);
150             if (allclosure.contains(cl)) {
151                 continu = false;
152             } else {
153                 allclosure.add(cl);
154             }
155         } while (continu);
156 
157         return allclosure;
158     }
159 
160     /**
161      * Returns the lecticaly next closed set of the specified one.
162      *
163      * This treatment is an implementation of the best knowm algorithm of Wille
164      * whose complexity is in O(Cl|S|^2), where S is the initial set of elements,
165      * and Cl is the closure computation complexity.
166      *
167      * @param   cl  a concept
168      *
169      * @return  the lecticaly next closed set
170      */
171     public Concept nextClosure(Concept cl) {
172         TreeSet<Comparable> set = new TreeSet(this.getSet());
173         boolean success = false;
174         TreeSet setA = new TreeSet(cl.getSetA());
175         Comparable ni = set.last();
176         do {
177             ni = (Comparable) set.last();
178             set.remove(ni);
179             if (!setA.contains(ni)) {
180                 setA.add(ni);
181                 TreeSet setB = this.closure(setA);
182                 setB.removeAll(setA);
183                 if (setB.isEmpty() || ((Comparable) setB.first()).compareTo(ni) >= 1) {
184                     setA = this.closure(setA);
185                     success = true;
186                 } else {
187                     setA.remove(ni);
188                 }
189             } else {
190                 setA.remove(ni);
191             }
192         } while (!success && ni.compareTo(this.getSet().first()) >= 1);
193         return new Concept(setA, false);
194     }

197    /**
198     * Returns the precedence graph of this component.
199     *
200     * Nodes of the graph are elements of this component.
201     * There is an edge from element a to element b when
202     * a belongs to the closure of b.
203     *
204     * The rule a -> a isn't added to the precedence graph
205     *
206     * When precedence graph is acyclic, then this component is a reduced one.
207     *
208     * @return  the precedence graph
209     */
210     public DGraph precedenceGraph() {
211         // compute a TreeMap of closures for each element of the component
212         TreeMap<Comparable, TreeSet<Comparable>> closures = new TreeMap<Comparable, TreeSet<Comparable>>();
213         for (Comparable x : this.getSet()) {
214             ComparableSet setX = new ComparableSet();
215             setX.add(x);
216             closures.put(x, this.closure(setX));
217         }
218         // nodes of the graph are elements
219         DGraph prec = new DGraph();
220         TreeMap<Comparable, Node> nodeCreated = new TreeMap<Comparable, Node>();
221         for (Comparable x : this.getSet()) {
222             Node n = new Node(x);
223             prec.addNode(n);
224             nodeCreated.put(x, n);
225         }
226         // edges of the graph are closures containments
227         for (Comparable from : this.getSet()) {
228             for (Comparable to : this.getSet()) {
229                 if (!from.equals(to)) {
230                     // check if from belongs to the closure of to
231                     if (closures.get(to).contains(from)) {
232                         prec.addEdge(nodeCreated.get(from), nodeCreated.get(to));
233                     }
234                 }
235             }
236         }
237         return prec;
238     }

240     /**
241      * This function returns all reducible elements.
242      *
243      * A reducible elements is equivalent by closure to one or more other attributes.
244      * Reducible elements are computed using the precedence graph of the closure system.
245      * Complexity is in O()
246      *
247      * @return  The map of reductible attributes with their equivalent attributes
248      */
249     public TreeMap<Object, TreeSet> getReducibleElements() {
250         // If you can't remove nodes, put them in the rubbish bin ...
251         TreeSet<Node> rubbishBin = new TreeSet<Node>();
252         // Initialise a map Red of reducible attributes
253         TreeMap<Object, TreeSet> red = new TreeMap();
254         // Initialise the precedence graph G of the closure system
255         DGraph graph = this.precedenceGraph();
256         // First, compute each group of equivalent attributes
257         // This group will be a strongly connected component on the graph.
258         // Then, only one element of each group is skipped, others will be deleted.
259         DAGraph cfc = graph.getStronglyConnectedComponent();
260         for (Node node : cfc.getNodes()) {
261             // Get list of node of this component
262             TreeSet<Node> sCC = (TreeSet<Node>) node.getContent();
263             if (sCC.size() > 1) {
264                 Node y = sCC.first();
265                 TreeSet yClass = new TreeSet();
266                 yClass.add(y.getContent());
267                 for (Node x : sCC) {
268                     if (!x.getContent().equals(y.getContent())) {
269                         rubbishBin.add(x); // instead of : graph.removeNode(x);
270                         red.put(x.getContent(), yClass);
271                     }
272                 }
273             }
274         }
275         // Next, check if an attribute is equivalent to emptyset
276         // i.e. its closure is equal to emptyset closure
277         TreeSet<Node> sinks = new TreeSet<Node>(graph.getSinks());
278         sinks.removeAll(rubbishBin);
279         if (sinks.size() == 1) {
280             Node s = sinks.first();
281             red.put(s.getContent(), new TreeSet());
282             rubbishBin.add(s); // instead of : graph.removeNode(s);
283         }
284         // Finaly, checking a remaining attribute equivalent to its predecessors or not may reduce more attributes.
285         // Check all remaining nodes of graph G
286         TreeSet<Node> remainingNodes = new TreeSet<Node>();
287         for (Node n : graph.getNodes()) {
288             remainingNodes.add(n);
289         }
290         remainingNodes.removeAll(rubbishBin);
291         for (Node x : remainingNodes) {
292             TreeSet<Node> predecessors = graph.getPredecessorNodes(x);
293             predecessors.removeAll(rubbishBin);
294             if (predecessors.size() > 1) {
295                 // Create the closure of x
296                 TreeSet set = new TreeSet();
297                 set.add(x.getContent());
298                 TreeSet closureSet = this.closure(set);
299                 // Create the closure of predecessors
300                 TreeSet<Comparable> pred = new TreeSet<Comparable>();
301                 for (Node n : predecessors) {
302                     pred.add((Comparable) n.getContent());
303                 }
304                 TreeSet<Comparable> closureP = this.closure(pred);
305                 // Check the equality of two closures
306                 if (closureSet.containsAll(closureP) && closureP.containsAll(closureSet)) {
307                     red.put(x.getContent(), pred);
308                 }
309             }
310         }
311         // Finally, return the list of reducible elements with their equivalent attributes.
312         return red;
313     }
