24  /**
25   * This class generates bijective components issued from lattice theory for a specified
26   * closure system.
27   *
28   * Bijective components are:
29   * closed set lattice or concept lattice, reduced lattice, reduced context, canonical direct basis, minimal generators
30   * and canonical basis, dependency graph.
31   *
32   * A closure system is described by the abstract class {@link ClosureSystem}.
33   * In this package, a closure system can be instancied by an implicational
34   * system described by class {@link ImplicationalSystem}) or a context described by
35   * class {@link Context}).
36   *
37   * This class provides a constructor, and only two methods: the method {@link #compute}
38   * generates all the bijective components of the specified closure system; and the method
39   * {@link #save} saves theses components in files.
40   *
41   * This class can be used as follows:
42   *
43   * ~~~Java
44   * BijectiveComponents bc = new BijectiveComponents(initialClosureSystem);
45   * bc.compute();
46   * bc.save(dirString,nameString);
47   * ~~~
48   *
49   * ![BijectiveComponents](BijectiveComponents.png)
50   *
51   * @uml BijectiveComponents.png
52   * !include resources/fr/kbertet/lattice/BijectiveComponents.iuml
53   * !include resources/fr/kbertet/lattice/ClosureSystem.iuml
54   *
55   * hide members
56   * show BijectiveComponents members
57   * class BijectiveComponents #LightCyan
58   * title BijectiveComponents UML graph
59   */

115     /**
116      * Initialise the closure system.
117      *
118      * @param   closureSystem  initial closure system
119      *
120      * @return  this for chaining
121      */
122     public BijectiveComponents initialise(ClosureSystem closureSystem) {
123         this.closureSystem = closureSystem;
124         this.lattice = null;
125         this.reducedLattice = null;
126         this.table = null;
127         this.dependencyGraph = null;
128         this.minimalGenerators = null;
129         this.canonicalDirectBasis = null;
130         this.canonicalBasis = null;
131         return this;
132     }

134     /**
135      * Generates all the bijective components included in this component
136      * issued from the initial closure system `closureSystem`.
137      *
138      * The closed set lattice is generated and obtained by
139      *
140      * ~~~Java
141      * this.getClosureSystem().lattice();
142      * ~~~
143      *
144      * The reduced lattice is obtained by
145      *
146      * ~~~Java
147      * this.getLattice().getIrreduciblesReduction();
148      * ~~~
149      *
150      * The reduced table is obtained by
151      *
152      * ~~~Java
153      * this.getReducedLattice().getTable();
154      * ~~~
155      *
156      * The dependency graph is obtained by
157      *
158      * ~~~Java
159      * this.getReducedLattice().getDependencyGraph();
160      * ~~~
161      *
162      * Minimal generators are obtained by
163      *
164      * ~~~Java
165      * this.getReducedLattice().getMinimalGenerators();
166      * ~~~
167      *
168      * The canonical direct basis is obtained by
169      *
170      * ~~~Java
171      * this.getReducedLattice().getCanonicalDirectBasis();
172      * ~~~
173      *
174      * The canonical basis is obtained by
175      *
176      * ~~~Java
177      * (new ImplicationalSystem(this.canonicalDirectBasis)).makeCanonicalBasis();
178      * ~~~
179      *
180      * @return  time of computation
181      */
182     public long compute() {
183         this.initialise(closureSystem);
184         long debut = new Date().getTime();
185         this.getLattice();
186         this.getReducedLattice();
187         this.getTable();
188         this.getDependencyGraph();
189         this.getMinimalGenerators();
190         this.getCanonicalDirectBasis();
191         this.getCanonicalBasis();
192         long fin = new Date().getTime();
193         return fin - debut;
194     }

196     /**
197      * Saves all the bijective components included in this component in files
198      * saved in the specified directory.
199      * A global description is saved in file `name+"Readme.txt"`.
200      *
201      * The specified name is used to defined a name for each file.
202      *
203      * @param   directory  location to save file
204      * @param   name       name of the files
205      *
206      * @throws  IOException  When an IOException occurs
207      */
208     public void save(String directory, String name) throws IOException {
209         // create the directory
210         directory = directory + File.separator + name + "BijectiveComponents" + File.separator;
211         File f = new File(directory);
212         f.mkdir();
213         directory += name;
214         BufferedWriter file = new BufferedWriter(new FileWriter(directory + "Readme.txt"));
215         // saves the inital closure system
216         String nameInit = directory + "InitialClosureSystem.txt";
217         this.closureSystem.save(nameInit);
218         file.write("-> Initial closure system saved in " + nameInit + ": \n");
219         file.write(this.closureSystem.toString() + "\n");
220         // saves the closed set lattice
221         String nameLattice = directory + "Lattice.dot";
222         this.getLattice().save(nameLattice);
223         file.write("-> Closed set or concept lattice saved in " + nameLattice + "\n");
224         // saves the reduced lattice
225         String nameReducedLattice = directory + "ReducedLattice.dot";
226         this.getReducedLattice().save(nameReducedLattice);
227         file.write("-> Reduced lattice saved in " + nameReducedLattice + "\n");
228         // saves the reduced table
229         String nameTable = directory + "Table.txt";
230         this.getTable().save(nameTable);
231         file.write("-> Table of the reduced lattice saved in " + nameTable + "\n");
232         file.write(this.table.toString() + "\n");
233         // saves the canonical basis
234         String nameCB = directory + "CanonicalBasis.txt";
235         this.getCanonicalBasis().save(nameCB);
236         file.write("-> Canonical basis saved in " + nameCB + ": \n");
237         file.write(this.canonicalBasis.toString() + "\n");
238         // saves the canonical direct basis
239         String nameCDB = directory + "CanonicalDirectBasis.txt";
240         this.getCanonicalDirectBasis().save(nameCDB);
241         file.write("-> Canonical direct basis of the reduced lattice saved in " + nameCDB + ": \n");
242         file.write(this.canonicalDirectBasis.toString() + "\n");
243         // saves the dependency graph
244         String nameODGraph = directory + "DependencyGraph.dot";
245         this.getDependencyGraph().save(nameODGraph);
246         file.write("-> Dependency Graph  of the reduced lattice saved in " + nameODGraph + " \n");
247         // saves the minimal generators
248         file.write("-> Minimal generators  of the reduced lattice are " + this.minimalGenerators + "\n");
249         file.close();
250     }
