28  /**
29   * This class extends class {@link fr.kbertet.dgraph.DAGraph} to provide specific methods to manipulate a lattice.
30   *
31   * A lattice is a directed acyclic graph ({@link fr.kbertet.dgraph.DAGraph}) containing particular nodes denoted join and meet\
32   * (a dag is a lattice if and only if each pair of nodes admits a join and a meet).
33   *
34   * Since checking the lattice property is very time-expensive, this property is not ensured
35   * for components of this class. However, it can be explicitely ckecked using
36   * method {@link #isLattice}.
37   *
38   * This class provides methods implementing classical operation on a lattice
39   * issued from join and meet operation and irreducibles elements,
40   * and methods that returns a condensed representation of a lattice.
41   *
42   * A well-known condensed representation of a lattice is its table, obtained by method
43   * {@link #getTable},
44   * where join-irreducibles are in column and meet-irreducibles are in rown.
45   *
46   * Another condensed representation is its dependency graph obtained by method
47   * {@link #getDependencyGraph}.
48   *
49   * The dependency graph is a directed graph where nodes are join-irreducibles,
50   * edges corresponds to the dependency relation between two join-irreducibles
51   * and are valuated by a family of subsets of irreducibles.
52   *
53   * The dependency graph encodes two other condensed representation of a lattice
54   * that are its minimal generators and its canonical direct basis
55   * that can be obtained from the dependency graph by methods
56   * {@link #getMinimalGenerators} and {@link #getCanonicalDirectBasis}.
57   *
58   * ![Lattice](Lattice.png)
59   *
60   * @todo  remove useless comments: Karell
61   *
62   * @uml Lattice.png
63   * !include resources/fr/kbertet/dgraph/DAGraph.iuml
64   * !include resources/fr/kbertet/dgraph/DGraph.iuml
65   * !include resources/fr/kbertet/dgraph/Edge.iuml
66   * !include resources/fr/kbertet/dgraph/Node.iuml
67   * !include resources/fr/kbertet/lattice/Lattice.iuml
68   *
69   * hide members
70   * show Lattice members
71   * class Lattice #LightCyan
72   * title Lattice UML graph
73   */

128    /**
129     * Check if this component is a lattice.
130     *
131     * There exists several caracterizations of a lattice.
132     * The characterization implemented is the following:
133     * A lattice is a DAG if there exists a meet for each pair of node,
134     * and a unique maximal node.
135     *
136     * This treatment is performed in O(n^3), where n is the number of nodes.
137     *
138     * @return  the truth value for this property
139     */
140    public boolean isLattice() {
141         if (!this.isAcyclic()) {
142             return false;
143         }
144         for (Node x : this.getNodes()) {
145             for (Node y : this.getNodes()) {
146                 if (this.meet(x, y) == null) {
147                     return false;
148                 }
149             }
150         }
151         return this.max().size() == 1;
152     }

154    /**
155     * Return true if this component is congruence normal.
156     *
157     * A lattice $L$ is in class CN (Congruence Normal) is there exists a sequence $L_1,\ldots,L_p$ of lattices with $L_p=L$,
158     * together with a sequence $C_1,\ldots,C_{p-1}$ such that $C_i$ is a convex of $L_i$ and $L_{i+1}$ is obtained
159     * by doubling the convex $C_i$ in $L_i$.
160     *
161     * See {@link fr.kbertet.lattice.LatticeFactory} for the doubling convex method which is not used here.
162     *
163     * This computation is done in O((|J|+|M|)^2|X|) from the transitive reduction of L.
164     *
165     * This recognition algorithm was first written in :
166     * "Doubling convex serts in lattices : characterizations and recognition algorithm", Bertet K., Caspard N. 2002.
167     *
168     * @return true if this component is congruence normal.
169     */
170    public boolean isCN() {
171        TreeSet<Node> joins = this.joinIrreducibles();
172        TreeSet<Node> meats = this.meetIrreducibles();
173        ArrowRelation arrows = new ArrowRelation(this);
174        Context dbl = arrows.getDoubleArrowTable();
175        // steps are connected component of the double arrow table.
176        ArrayList<Concept> steps = new ArrayList<Concept>();
177        while (!joins.isEmpty()) {
178            TreeSet<Comparable> setA = new TreeSet<Comparable>();
179            TreeSet<Comparable> setB = new TreeSet<Comparable>();
180            int cardA = setA.size();
181            setA.add(joins.first());
182            while (cardA != setA.size()) { // something added
183                cardA = setA.size(); // update card
184                for (Comparable c : setA) {
185                    setB.addAll(dbl.getIntent(c));
186                }
187                for (Comparable c : setB) {
188                    setA.addAll(dbl.getExtent(c));
189                }
190            }
191            steps.add(new Concept(setA, setB));
192            joins.removeAll(setA);
193        }
194        for (Concept c : steps) {
195            if (c.getSetB().isEmpty()) { // to be verified :-)
196                return false;
197            }
198            for (Comparable j : c.getSetA()) {
199                for (Comparable m : c.getSetB()) {
200                    if ((arrows.getEdge((Node) j, (Node) m).getContent() != "UpDown")
201                            && (arrows.getEdge((Node) j, (Node) m).getContent() != "Circ")) {
202                        return false;
203                    }
204                }
205            }
206        }
207        joins = this.joinIrreducibles();
208        meats = this.meetIrreducibles();
209        DGraph phi = new DGraph();
210        for (Node j : joins) {
211            for (Node m : meats) {
212                int indJ = 0; // Search for the step containning j
213                while ((indJ < steps.size()) && (!steps.get(indJ).containsInA(j))) {
214                    indJ++;
215                }
216                if ((phi.getNodeByContent(indJ) == null) && (indJ != steps.size())) {
217                    phi.addNode(new Node(indJ));
218                }
219                int indM = 0; // Search for the step containning m
220                while ((indM < steps.size()) && (!steps.get(indM).containsInB(m))) {
221                    indM++;
222                }
223                if ((phi.getNodeByContent(indM) == null) && (indM != steps.size())) {
224                    phi.addNode(new Node(indM));
225                }
226                if (indM != steps.size() && indJ != steps.size()) {
227                    if (arrows.getEdge((Node) j, (Node) m).getContent() == "Up") {
228                        phi.addEdge(phi.getNodeByContent(indM), phi.getNodeByContent(indJ));
229                    }
230                    if (arrows.getEdge((Node) j, (Node) m).getContent() == "Down") {
231                        phi.addEdge(phi.getNodeByContent(indJ), phi.getNodeByContent(indM));
232                    }
233                }
234            }
235        }
236        return (phi.isAcyclic());
237    }
238    /**
239     * Returns true if this component is an atomistic lattice.
240     *
241     * A lattice is atomistic if its join irreductibles are atoms (e.g. successors of bottom)
242     *
243     * @return true if this component is atomistic.
244     */
245    public boolean isAtomistic() {
246        TreeSet<Node> join = this.joinIrreducibles();
247        TreeSet<Node> atoms = new TreeSet<Node>(this.getSuccessorNodes(this.bottom()));
248        return join.containsAll(atoms) && atoms.containsAll(join);
249    }
250    /**
251     * Returns true if this component is an coatomistic lattice.
252     *
253     * A lattice is coatomistic if its mett irreductibles are coatoms (e.g. predecessors of top)
254     *
255     * @return true if this component is coatomistic.
256     */
257    public boolean isCoAtomistic() {
258        TreeSet<Node> meet = this.meetIrreducibles();
259        TreeSet<Node> coatoms = this.getPredecessorNodes(this.top());
260        return meet.containsAll(coatoms) && coatoms.containsAll(meet);
261    }

290     /**
291      * Returns the meet of the two specified nodes if it exists.
292      *
293      * @param   x  the first node
294      * @param   y  the second node
295      *
296      * @return  the node which is at the meet of the nodes or null if it does not exist
297      */
298     public Node meet(Node x, Node y) {
299         SortedSet<Node> xMinorants = this.minorants(x);
300         xMinorants.add(x);
301 
302         SortedSet<Node> yMinorants = this.minorants(y);
303         yMinorants.add(y);
304 
305         xMinorants.retainAll(yMinorants);
306         DAGraph graph = this.getSubgraphByNodes(xMinorants);
307         TreeSet<Node> meet = new TreeSet<Node>(graph.max());
308         if (meet.size() == 1) {
309             return meet.first();
310         }
311         return null;
312     }

314     /**
315      * Returns the join of the two specified nodes if it exists.
316      *
317      * @param   x  the first node
318      * @param   y  the second node
319      *
320      * @return  the node which is at the join of the nodes or null if it does not exist
321      */
322     public Node join(Node x, Node y) {
323         SortedSet<Node> xMajorants = this.majorants(x);
324         xMajorants.add(x);
325 
326         SortedSet<Node> yMajorants = this.majorants(y);
327         yMajorants.add(y);
328 
329         xMajorants.retainAll(yMajorants);
330         DAGraph graph = this.getSubgraphByNodes(xMajorants);
331         TreeSet<Node> join = new TreeSet<Node>(graph.min());
332         if (join.size() == 1) {
333             return join.first();
334         }
335         return null;
336     }

340     /**
341      * Returns the set of join irreducibles of this component.
342      *
343      * Join irreducibles are nodes with an unique immediate predecessor
344      * in the transitive and reflexive reduction.
345      * This component is first reduced reflexively and transitively.
346      *
347      * @return  the set of join irreducibles of this component
348      */
349     public TreeSet<Node> joinIrreducibles() {
350         DAGraph graph = new DAGraph(this);
351         graph.reflexiveReduction();
352         graph.transitiveReduction();
353         TreeSet<Node> set = new TreeSet();
354         for (Node n : graph.getNodes()) {
355             if (graph.getPredecessorNodes(n).size() == 1) {
356                 set.add(n);
357             }
358         }
359         return set;
360     }

444     /**
445      * Generates and returns the isomorphic closed set lattice defined on the join irreducibles set.
446      *
447      * Each node of this component is replaced  by a node containing its join irreducibles
448      * predecessors stored in a closed set.
449      *
450      * @return  the isomorphic closed set lattice defined on the join irreducibles set
451      */
452     public ConceptLattice joinClosure() {
453         ConceptLattice csl = new ConceptLattice();
454         // associates each node to a new closed set of join irreducibles
455         TreeSet<Node> join = this.joinIrreducibles();
456         TreeMap<Node, Concept> closure = new TreeMap<Node, Concept>();
457         Lattice lattice = new Lattice(this);
458         lattice.transitiveClosure();
459         lattice.reflexiveClosure();
460         for (Node to : lattice.getNodes()) {
461             ComparableSet jx = new ComparableSet();
462             for (Node from : lattice.getPredecessorNodes(to)) {
463                 if (join.contains(from)) {
464                     jx.add(from.getContent());
465                 }
466             }
467             closure.put(to, new Concept(jx, false));
468         }
469         // addition of nodes
470         for (Node n : this.getNodes()) {
471             csl.addNode(closure.get(n));
472         }
473         // addition of edges
474         for (Edge ed : this.getEdges()) {
475             csl.addEdge(closure.get(ed.getFrom()), closure.get(ed.getTo()));
476         }
477         return csl;
478     }

560     /**
561      * Returns the smallest set of nodes of this component containing S such that if a,b in JS then join(a,b) in JS.
562      *
563      * @param s set of nodes to be closed
564      * @return (JS) join-closure of s
565      */
566     public ComparableSet joinClosure(ComparableSet s) {
567         // Algorithm is true because join is idempotent & commutative
568         ComparableSet stack = new ComparableSet();
569         stack.addAll(s); // Nodes to be explored
570         ComparableSet result = new ComparableSet();
571         while (!stack.isEmpty()) {
572             Node c = (Node) stack.first();
573             stack.remove(c);
574             result.add(c);
575             Iterator<Node> it = stack.iterator();
576             ComparableSet newNodes = new ComparableSet(); // Node to be added. Must be done AFTER while
577             while (it.hasNext()) {
578                 Node n = this.join(it.next(), c);
579                 if (!result.contains(n) && !stack.contains(n)) {
580                     newNodes.add(n);
581                 }
582             }
583             stack.addAll(newNodes);
584         }
585         return result;
586     }
587     /**
588      * Returns the smallest set of nodes of this component containing S such that if a,b in MS then meet(a,b) in MS.
589      *
590      * @param s set of nodes to be closed
591      * @return (MS) meet-closure of s
592      */
593     public ComparableSet meetClosure(ComparableSet s) {
594         // Algorithm is true because meet is idempotent & commutative
595         ComparableSet stack = new ComparableSet();
596         stack.addAll(s); // Nodes to be explored
597         ComparableSet result = new ComparableSet();
598         while (!stack.isEmpty()) {
599             Node c = (Node) stack.first();
600             stack.remove(c);
601             result.add(c);
602             Iterator<Node> it = stack.iterator();
603             ComparableSet newNodes = new ComparableSet(); // Node to be added. Must be done AFTER while
604             while (it.hasNext()) {
605                 Node n = this.meet(it.next(), c);
606                 if (!result.contains(n) && !stack.contains(n)) {
607                     newNodes.add(n);
608                 }
609             }
610             stack.addAll(newNodes);
611         }
612         return result;
613     }
614     /**
615      * Returns the smallest sublattice of this component containing s.
616      *
617      * @param s set of nodes to be closed.
618      * @return the smallest sublattice of this component containing s.
619      */
620     public ComparableSet fullClosure(ComparableSet s) {
621         ComparableSet result = new ComparableSet();
622         result.addAll(s);
623         int present = result.size();
624         int previous = 0;
625         while (previous != present) {
626             previous = present;
627             result = this.joinClosure(result);
628             result = this.meetClosure(result);
629             present = result.size();
630         }
631         return result;
632     }
633     /**
634      * Returns the list of all sets of nodes that generates all nodes.
635      * Both join and meet operations are allowed and
636      * the sets are minimal for inclusion.
637      *
638      * @return : List of all hybridGenerators families.
639      */
640     public TreeSet<ComparableSet> hybridGenerators() {
641         TreeSet<Node> joinIrr = this.joinIrreducibles();
642         TreeSet<Node> meetIrr = this.meetIrreducibles();
643         ComparableSet bothIrr = new ComparableSet();
644         for (Node n : joinIrr) {
645             if (meetIrr.contains(n)) {
646                 bothIrr.add(n);
647             }
648         } // bothIrr contains nodes that are join and meet irreductibles.
649 
650         TreeSet<ComparableSet> generators = new TreeSet<ComparableSet>();
651         // First point is that all minimal families have the same number of nodes.
652         LinkedList<ComparableSet> list = new LinkedList<ComparableSet>(); // Family of sets to be examined
653         list.add(bothIrr);
654         while (!list.isEmpty()) {
655             int test;
656             if (generators.isEmpty()) {
657                 test = this.sizeNodes();
658             } else {
659                 test = generators.first().size();
660             }
661             if (test < list.peek().size()) {
662                 // Elements are sorted by size, thus if the first is to large, all are.
663                 list.clear();
664             } else {
665                 ComparableSet family = list.poll(); // Retrieve and remove head.
666                 if (this.fullClosure(family).size() == this.sizeNodes()) {
667                     // This family generates l
668                     generators.add(family);
669                 } else {
670                     for (Node n : this.getNodes()) {
671                         ComparableSet newFamily = new ComparableSet();
672                         newFamily.addAll(family);
673                         newFamily.add(n);
674                         if (!list.contains(newFamily)) {
675                             list.add(newFamily); // add at the end, bigger families
676                         }
677                     }
678                 }
679             }
680         }
681         return generators;
682     }

784    /** Returns the dependency graph of this component.
785     *
786     * The dependency graph is a condensed representation of a lattice that encodes
787     * its minimal generators, and its canonical direct basis.
788     *
789     * In the dependency graph, nodes are join irreducibles,
790     * egdes correspond to the dependency relation between join-irreducibles
791     * (j -> j' if and only if there exists a node x in the lattice such that
792     * x not greather than j and j', and x v j' > j),
793     * and edges are labeled with the smallest subsets X of join-irreducibles such that
794     * the join of elements of X corresponds to the node x of the lattice.
795     *
796     * The dependency graph could has been already computed in the case where this component
797     * has been instancied as the diagramm of the closed set lattice of a closure system
798     * using the static method
799     * {@link ConceptLattice#diagramLattice}
800     * This method implements an adaptation adaptation of Bordat's
801     * where the dependency graph is computed while the lattice is generated.
802     *
803     * However, it is generated in O(nj^3) where n is the number of nodes of the lattice, and
804     * j is the number of join-irreducibles of the lattice.
805     *
806     * @return  the dependency graph
807     */
808     public DGraph getDependencyGraph() {
809         if (!(this.dependencyGraph == null)) {
810             return this.dependencyGraph;
811         }
812         this.dependencyGraph = new DGraph();
813         // nodes of the dependency graph are join-irreducibles
814         TreeSet<Node> joins = this.joinIrreducibles();
815         for (Node j : joins) {
816             this.dependencyGraph.addNode(j);
817         }
818         // computes the transitive closure of the join-irreducibles subgraph of this compnent
819         DAGraph joinG = this.irreduciblesSubgraph();
820         joinG.transitiveClosure();
821         // edges of the dependency graph are dependency relation between join-irreducibles
822         // they are first valuated by nodes of the lattice
823         for (Node j1 : joins) {
824             SortedSet<Node> majj1 = this.majorants(j1);
825             for (Node j2 : joins) {
826                 if (!j1.equals(j2)) {
827                     // computes the set S of nodes not greather than j1 and j2
828                     TreeSet<Node> set = new TreeSet<Node>(this.getNodes());
829                     set.removeAll(majj1);
830                     set.removeAll(this.majorants(j2));
831                     set.remove(j1);
832                     set.remove(j2);
833                     for (Node x: set) {
834                         // when j2 V x greather than j1 then add a new edge from j1 to J2
835                         // or only a new valuation when the edge already exists
836                         if (majj1.contains(this.join(j2, x))) {
837                             Edge ed = this.dependencyGraph.getEdge(j1, j2);
838                             if (ed == null) {
839                                 ed = new Edge(j1, j2, new TreeSet<ComparableSet>());
840                                 this.dependencyGraph.addEdge(ed);
841                             }
842                             // add {Jx minus predecessors in joinG of j in Jx} as valuation of edge
843                             // from j1 to j2
844                             TreeSet<ComparableSet> valEd = (TreeSet<ComparableSet>) ed.getContent();
845                             ComparableSet newValByNode = new ComparableSet(this.joinIrreducibles(x));
846                             for (Comparable j : this.joinIrreducibles(x)) {
847                                 newValByNode.removeAll(joinG.getPredecessorNodes((Node) j));
848                             }
849                             ComparableSet newVal = new ComparableSet();
850                             for (Object j : newValByNode) {
851                                 Node n = (Node) j;
852                                 newVal.add(n.getContent());
853                             }
854                             ((TreeSet<ComparableSet>) ed.getContent()).add(newVal);
855                             // Minimalisation by inclusion of valuations on edge j1->j2
856                             valEd = new TreeSet<ComparableSet>((TreeSet<ComparableSet>) ed.getContent());
857                             for (ComparableSet x1 : valEd) {
858                                 if (x1.containsAll(newVal) && !newVal.containsAll(x1)) {
859                                     ((TreeSet<ComparableSet>) ed.getContent()).remove(x1);
860                                 }
861                                 if (!x1.containsAll(newVal) && newVal.containsAll(x1)) {
862                                     ((TreeSet<ComparableSet>) ed.getContent()).remove(newVal);
863                                 }
864                             }
865                         }
866                     }
867                  }
868              }
869          }
870         // minimalisation of edge's content to get only inclusion-minimal valuation for each edge
871         /**for (Edge ed : this.dependencyGraph.getEdges()) {
872              TreeSet<ComparableSet> valEd = new TreeSet<ComparableSet>(((TreeSet<ComparableSet>)ed.getContent()));
873              for (ComparableSet X1 : valEd)
874                  for (ComparableSet X2 : valEd)
875                      if (X1.containsAll(X2) && !X2.containsAll(X1))
876                          ((TreeSet<ComparableSet>)ed.getContent()).remove(X1);
877         }**/
878         return this.dependencyGraph;
879     }

902     /**
903      * Returns the canonical direct basis of the lattice.
904      *
905      * The canonical direct basis is a condensed representation of a lattice encoding
906      * by the dependency graph.
907      *
908      * This canonical direct basis is deduced from the dependency graph of the lattice:
909      * for each edge b -> a valuated by a subset X, the rule a+X->b is a rule of the
910      * canonical direct basis.
911      *
912      * If not yet exists, the dependency graph of this component has to be generated
913      * by method {@link #getDependencyGraph}.
914      *
915      * @return  the canonical direct basis of the lattice
916      */
917     public ImplicationalSystem getCanonicalDirectBasis() {
918         DGraph odGraph = this.getDependencyGraph();
919         // initialise elements of the ImplicationalSystem with nodes of the ODGraph
920         ImplicationalSystem bcd = new ImplicationalSystem();
921         for (Node n : odGraph.getNodes()) {
922             bcd.addElement((Comparable) n.getContent());
923         }
924         // computes rules of the BCD from edges of the ODGraph
925         for (Edge ed : odGraph.getEdges()) {
926             Node from = ed.getFrom();
927             Node to = ed.getTo();
928             TreeSet<ComparableSet> l = (TreeSet<ComparableSet>) ed.getContent();
929             for (ComparableSet set : l) {
930                 ComparableSet premise = new ComparableSet(set);
931                 premise.add((Comparable) to.getContent());
932                 ComparableSet conclusion = new ComparableSet();
933                 conclusion.add((Comparable) from.getContent());
934                 bcd.addRule(new Rule(premise, conclusion));
935             }
936         }
937         //bcd.makeLeftMinimal();
938         bcd.makeCompact();
939         return bcd;
940     }
